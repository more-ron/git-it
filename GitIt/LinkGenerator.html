def initialize(repository_url)
  @repository_url = repository_url

  @repo = %r^git@(?<site>.*):(?<user>.*)\/(?<proj>.*).git$/.match(@repository_url)

  @site = @repo[:site]
  @user = @repo[:user]
  @proj = @repo[:proj]
end
